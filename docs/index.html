About ATRC

ATRC is a powerful configuration library that goes beyond simple key-value parsing. It features a smart preprocessor with conditional logic, file inclusion capabilities, and dynamic variable injection. Whether you're building game engines, multimedia applications, or any software requiring sophisticated configuration management, ATRC provides the tools you need. The library supports both C++17 and C interfaces, making it suitable for modern applications while maintaining compatibility with legacy systems. With features like platform detection, architecture awareness, and advanced string handling including raw strings and enums, ATRC adapts to your project's specific requirements.

Example Configuration File

#!ATRC
# Include global definitions
#.INCLUDE globals.atrc

# Global variables
%user%=admin
%env%=production

# Platform detection
#.IF WINDOWS
Platform=Windows
#.ELSE
Platform=Unix
#.ENDIF

[ServerConfig]
Host=%user%@server.com
Environment=%env%

[Injection Example]
# Left-to-right injection
Inject1=Values: %*%, %*%, %*%
# Ordered injection: 0=A, 1=B, 2=C â†’ displays C, A, B
Inject2=Ordered: %*2%, %*0%, %*1%

[EnumExample]
Status=Active    # GetEnumValue() returns 0 for first occurrence

[RawStringExample]
#.SR KEY
MultiLineConfig=This is a
multi-line configuration
that preserves formatting!
#.ER

Example C++ Program

// MAIN.CPP
#include <iostream>
#include <ATRC.h>

int main() {
    atrc::ATRC_FD fd("./config.atrc", ATRC_READ_ONLY);
    
    if (!fd.CheckStatus()) {
        std::cout << "File parsed unsuccessfully!" << std::endl;
        return 1;
    }

    // Access variables and keys
    std::cout << "User: " << fd["user"] << std::endl;
    std::cout << "Host: " << fd["ServerConfig"]["Host"] << std::endl;

    // Variable injection example
    std::string template_str = fd["Injection Example"]["Inject1"];
    std::vector<std::string> values = {"first", "second", "third"};
    std::string result = fd.InsertVar_S(template_str, values);
    std::cout << "Injected: " << result << std::endl;

    // Enum example
    int enumVal = fd.GetEnumValue("EnumExample", "Status");
    std::cout << "Enum value: " << enumVal << std::endl; // Returns 0

    // Raw string example
    std::string multiLine = fd["RawStringExample"]["MultiLineConfig"];
    std::cout << "Raw string:\n" << multiLine << std::endl;
    
    return 0;
}

Example C Program

// MAIN.C
#include "ATRC.h"
#include <stdio.h>

int main() {
    C_PATRC_FD filedata = Create_Empty_ATRC_FD();
    if (filedata == NULL) {
        printf("[FAIL] Create_Empty_ATRC_FD: Failed to create ATRC_FD\n");
        return 1;
    }

    if (!Read(filedata, "config.atrc", ATRC_READ_ONLY)) {
        printf("[FAIL] Read: Failed to read file 'config.atrc'\n");
        Destroy_ATRC_FD(filedata);
        return 1;
    }

    // Read variables
    const char* user = ReadVariable(filedata, "user");
    if (user != NULL) {
        printf("[PASS] User: %s\n", user);
    }

    // Read block keys
    const char* host = ReadKey(filedata, "ServerConfig", "Host");
    if (host != NULL) {
        printf("[PASS] Host: %s\n", host);
    }

    // Add new configuration
    if (AddBlock(filedata, "NewSection")) {
        printf("[PASS] AddBlock: Section added successfully\n");
        
        if (AddKey(filedata, "NewSection", "Setting", "Value")) {
            printf("[PASS] AddKey: Key added successfully\n");
        }
    }

    // Check existence
    if (DoesExistBlock(filedata, "ServerConfig")) {
        printf("[PASS] ServerConfig block exists\n");
    }

    Destroy_ATRC_FD(filedata);
    return 0;
}
